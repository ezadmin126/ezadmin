SELECT
    t.id,
    t.name,
    t.status,
    CASE
        WHEN t.status = 1 THEN '启用'
        WHEN t.status = 0 THEN '禁用'
        ELSE '未知'
    END as status_text,
    t.create_time,
    t.update_time,
    t.remark
FROM
    test_table t
WHERE
    t.delete_flag = 0
    {name?AND t.name LIKE '%{name}%'}
    {status?AND t.status = {status}}
    {create_time_start?AND t.create_time >= '{create_time_start}'}
    {create_time_end?AND t.create_time <= '{create_time_end}'}
